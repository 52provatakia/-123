        <button onclick="start_game('Σωματική Υγεία')">Σωματική Υγεία</button>

    <div id="score"></div>

    <py-script>

        import random

        from js import Audio, document

        # Ερωτήσεις ανά κατηγορία

        physical_health = [

            {"question": "Ποιο τρόφιμο είναι πιο υγιεινό;", 

             "options": [

                 {"text": "Μήλο", "image": "https://www.example.com/apple.png"},

                 {"text": "Πίτσα", "image": "https://www.example.com/pizza.png"}

             ],

             "answer": "Μήλο"

            },

            {"question": "Τι βοηθά στη διατήρηση της φυσικής κατάστασης;", 

             "options": [

                 {"text": "Περπάτημα", "image": "https://www.example.com/walking.png"},

                 {"text": "Καθισιό", "image": "https://www.example.com/sitting.png"}

             ],

             "answer": "Περπάτημα"

            }

        ]

        mental_health = [

            {"question": "Ποια δραστηριότητα μειώνει το άγχος;", 

             "options": [

                 {"text": "Διαλογισμός", "image": "https://www.example.com/meditation.png"},

                 {"text": "Τηλεόραση", "image": "https://www.example.com/tv.png"}

             ],

             "answer": "Διαλογισμός"

            },

            {"question": "Ποια είναι η καλύτερη ώρα για ύπνο;", 

             "options": [

                 {"text": "10:00 μ.μ.", "image": "https://www.example.com/bedtime.png"},

                 {"text": "2:00 π.μ.", "image": "https://www.example.com/night.png"}

             ],

             "answer": "10:00 μ.μ."

            }

        ]

        # Παγκόσμιες μεταβλητές

        questions = []

        score = 0

        correct_sound = Audio("https://www.example.com/correct-sound.mp3")

        incorrect_sound = Audio("https://www.example.com/incorrect-sound.mp3")

        # Ξεκινάμε το παιχνίδι

        def start_game(category):

            global questions, score

            score = 0

            if category == "Σωματική Υγεία":

                questions = physical_health

            elif category == "Ψυχική Υγεία":

                questions = mental_health

            Element("category-selector").write("")

            next_question()

        # Τυχαία ερώτηση

        def next_question():

            global questions

            if len(questions) == 0:

                show_summary()

                return

            question = random.choice(questions)

            questions.remove(question)

            display = f"<p>{question['question']}</p><div>"

            for option in question['options']:

                display += f"""

                    <button class="icon-button" onclick='check_answer("{option['text']}", 
"{question['answer']}", this)'>

                        <img src="{option['image']}" alt="{option['text']}">

                        <p>{option['text']}</p>

                    </button>

                """

            display += "</div>"

            Element("game-container").write(display)

        # Έλεγχος της απάντησης

        def check_answer(selected, correct, button):

            global score

            if selected == correct:

                score += 1

                correct_sound.play()

                button.className += " correct"

            else:

                incorrect_sound.play()

                button.className += " incorrect"

            Element("score").write(f"Σκορ: {score}")

            setTimeout(next_question, 1500)

        # Τελική περίληψη

        def show_summary():

            Element("game-container").write(f"<h2>Τέλος παιχνιδιού!</h2><p>Τελικό Σκορ: 
{score}</p>")
